## this

- 함수 호출 맥락
- 어떻게 함수를 호출하느냐에 따라 this가 가리키는 대상이 달라짐
- 함수와 객체를 연결해주는 실질적인 연결점의 역할

```jsx
function func(){
    if(window === this){

    }
}

var o = {
    func : function(){
        if(o === this){
            document.write("o === this");
        }
    }

o.func();

var funcThis = null;

    funcThis = this;

var o1 = Func();
if(funcThis === window){
    document.write('window <br />'); // window
}
 
var o2 = new Func();
if(funcThis === o2){
    document.write('o2 <br />'); // o2
}
```

## apply, call, bind

- 둘 다 함수의 this 명시적으로 바인딩할 때 사용됨
- 다른 객체의 파라미터나 메소드를 자기 것처럼 사용할 수 있음.
- call()
    - 첫 번째 매개변수로 thisArg 넘겨주고 파라미터 그대로 사용
    - fun.call(thisArg[arg1[, arg2[, ... ]]])
    - apply() 와 유사하지만 call() 함수에 전달될 인수 리스트를 받음

```jsx
var name = '영희'; 
var age = '19'; 
function Obj(name, age){ 
	this.name = name; this.age = age; 
} 
Obj.prototype.print = function(){
 console.log('name : '+this.name+' age : '+this.age); 
} 
var student1 = new Obj('철수', '20'); 
// 함수 내부의 this를 전역의 this 객체로 변경
student1.print.call(this); // 'name: 영희 age:19' 출력
```

- apply()
    - 주어진 this 값과 배열로 제공되는 arguments로 함수 호출
    - call() 과 유사하지만 apply() 함수는 인수들의 단일 배열을 받음
    - fun.apply(thisArg, [argsArray])

```jsx
// 매개 변수 전달 방식이 배열임
// 값은 call이랑 같음
student1.print.apply(this, ['male', 'A']);
```

```jsx
var o = {}
var p = {}
function func(){
    switch(this){
        case o:
            document.write('o<br />');
            break;
        case p:
            document.write('p<br />');
            break;
        case window:
            document.write('window<br />');
            break;          
    }
}
func(); // window
func.apply(o); // o
func.apply(p); // p
```

- bind()
    - 새로운 함수를 만들어 리턴함
    - bind 사용 시 바로 함수가 호출되지 않음

```jsx
// 새로운 함수 만들어서 리턴
var print = student1.print.bind(this, 'female','O'); 
// print 변수로 받음
print();
```

# 상속

- 객체의 로직을 그대로 물려 받는 또 다른 객체 만드는 기능

```jsx
function Person(name){
    this.name = name;

Person.prototype.name=null;
Person.prototype.introduce = function(){
    return 'My name is '+this.name; 
}
var p1 = new Person('egoing');
document.write(p1.introduce()+"<br />");

// 생성자 만듬

    this.name = name;

Programmer.prototype.coding = function(){
    return "hello world";
}
 
var p1 = new Programmer('egoing');
// Person이 가지고 있지 않은 기능 coding() 사용 가능
document.write(p1.coding()+"<br />");
```

# Object

- 객체의 가장 기본적인 형태 가지고 있음
- 아무것도 상속받지 않는 순수한 객체
- 모든 객체는 object 객체를 상속받는다.
- object 객체를 확장하면 모든 객체가 접근할 수 있는 API 를 만들 수 있다.
    - 다른 객체들에게도 영향을 주기 때문에 권장하지 않는다.

```jsx
Object.prototype.contain = function(neddle) {
    for(var name in this){
        if(this[name] === neddle){
            return true;
        }
    }
    return false;
}
var o = {'name':'egoing', 'city':'seoul'}
console.log(o.contain('egoing'));
var a = ['egoing','leezche','grapittie'];
console.log(a.contain('leezche'));
```
