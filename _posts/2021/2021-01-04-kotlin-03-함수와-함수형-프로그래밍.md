### 함수

- 여러 인자를 입력받아 기능을 수행하고 결괏값을 반환하는 코드의 모음
- 코드를 재사용하기 위해 사용됨

```kotlin
// : Int (반환값 자료형 명시)
fun sum(a: Int, b: Int): Int {  
	var sum = a + b
	return sum
}

// 간략화할 수 있다
fun sum(a: Int, b: Int) = a + b
fun sum(a: Int, b: Int): Int = a + b 
```

- Unit : 반환값이 없을 때 사용

```kotlin
fun printSum(a: Int, b: Int): Unit {
	println("sum of $a and $b is ${a+b}")
}

// Unit 생략
fun printSum(a: Int, b: Int) {
	println("sum of $a and $b is ${a+b}")
}
```

- default : 매개변수의 기본값 기능 제공

```kotlin
fun add(name: String, email: String = "default"){

}

add("이은혁") // email 자동으로 default 저장됨
add(name = "이은혁", email = "eunhyeok@gmail.com") // 매개변수 이름과 함께 호출 가능
```

- vararg : 가변인자 = 인자의 개수가 정해지지 않은 인자

```kotlin
fun normalVarargs(vararg counts: Int) {

}

normalVarargs(1, 2, 3, 4) // 정해지지 않음
```

### 함수형 프로그래밍

- 순수 함수를 작성하여 프로그램의 부작용을 줄이는 프로그래밍 기법
- 람다식, 고차 함수 사용
- 순수 함수 : 같은 인자에 대하여 항상 같은 값을 반환하는 함수 + 외부의 어떤 상태도 바꾸지 않는 함수
- 람다식 : 람다 대수에서 유래한 함수
    - 이름이 없는 함수로 2개 이상의 입력을 1개의 출력으로 단순화하는 개념

```kotlin
{x, y -> x + y}
```

- 고차 함수 : 다른 함수를 인자로 사용하거나 함수를 결괏값으로 반환하는 함수

```kotlin
fun main() {
	println(highFunc({x, y -> x + y}, 10, 20))
}

fun highFunc(sum: (Int, Int) -> Int, a: Int, b: Int): Int = sum (a, b)
```

- 람다식 선언과 할당

```kotlin
// 변수를 함수명처럼 사용 가능 // (Int, Int) -> 람다식의 자료형 선언 
// x: Int, y: Int 람다식의 매개변수
// x * y 람다식의 처리 내용
val multi: (Int, Int) -> Int = {x: Int, y: Int -> x * y}
val multi = {x: Int, y: Int -> x * y} // 선언 자료형 생략
val multi: (Int, Int) -> Int = {x, y -> x * y} // 매개변수 자료형 생략

// 매개변수가 없을 경우
val greet: () -> Unit = {println("Hello World!")} 
------------------------------------------------------
fun noParam(out: () -> String) = println(out())
noParam({"Hello World!"})

#####################################################
// 매개변수가 하나만 있을 경우
val square: (Int) -> Int = {x -> x * x} 
------------------------------------------------------
oneParam({a -> "Hello World! $a"})
oneParam{a -> "Hello World! $a"}
oneParam{"Hello World! $it"}

fun oneParam(out: (String) -> String) {
	println(out("OneParam"))
}

#####################################################
// 매개변수가 2개 이상일 경우
moreParam{ a, b -> "Hello World! $a $b"}

fun moreParam(out: (String, String) -> String) {
	println(out("OneParam", "TwoParam"))
}

#####################################################
// 일반 매개변수와 람다식 매개변수를 같이 사용하기

withArgs("Arg1", "Arg2", { a, b -> "Hello World! $a $b"})

fun withArgs(a: String, b: String, out: (String, String) -> String) {
	println(out(a, b))
}

#####################################################
// 일반 함수에 람다식 매개변수를 2개 이상 사용하기

twoLambda({a, b -> "First $a $b" }, {"Second $it"})

fun twoLambda(first: (String, String) -> String, second: (String) -> String) {
	println(first("OneParam", "TwoParam"))
	println(second("OneParam"))
}

#####################################################
fun funcParam(a: Int, b: Int, c: (Int, Int) -> Int): Int {
	return c(a, b)
}

fun sum(a: Int, b: Int) = a + b

funcParam(3, 2, ::sum) // 소괄호와 인자 생략 가능
```

- 제네릭 함수 : 데이터 자료형을 일반화해서 어떤 자료형이던 지정할 수 있는 자료형

```kotlin
fun <T> lock(reLock: ReetrantLock, body: () -> T): T {
	relock.lock()
}
```

### 코틀린 함수

- inline Function : 함수가 호출되는 곳에 함수 본문의 내용을 모두 복사해 넣어 함수의 분기 없이 처리되기 때문에 코드의 성능을 높이는데 사용됨

```kotlin
inline fun shortFunc(a: Int, out: (Int) -> Unit) {
}
```

- noinline : 일부 람다식을 인라인되지 않게 하기 위해 사용됨

```kotlin
inline fun sub(out1: () -> Unit, noinline out2: () -> Unit) {
}
```

- 확장 함수 : 기존의 클래스에 새로운 함수 추가

```kotlin
fun String.getLongString(target: String): String = 
	if(this.length > target.length) this else target
// String 클래스에 확장 함수 추가
```

- 중위 함수 : 클래스의 멤버를 호출할 때 사용하는 점을 생략하고 함수 이름 뒤에 소괄호를 붙이지 않아 직관적인 이름을 사용할 수 있는 표현법 = 일종의 연산자를 구현할 수 있다
    - 조건 : 멤버 메서드 혹은 확장함수 / 하나의 매개변수 필요 / infix 키워드 이용 정의

```kotlin
infix fun Int.multiply(x: Int): Int {
	return this * x
}

val multi = 3 mutliply 10
```

- tailrec : 스택에 계속 쌓이는 방식이 아닌 꼬리를 무는 형태로 반복 (스택 오버플로 방지) → 계산을 먼저 하고 재귀 함수가 호출됨

```kotlin
tailrec fun factorial(n: Int, run: Int = 1): Long {
	return if (n==1) run.toLong() else factorial(n-1, run*n)
}
```
