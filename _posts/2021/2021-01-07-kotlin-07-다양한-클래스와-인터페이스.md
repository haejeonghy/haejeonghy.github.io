### 추상 클래스와 인터페이스

- 추상(구체적이지 않은 것) 클래스 : 선언 등의 대략적인 설계 명세와 공통의 기능을 구현한 클래스
    - 인터페이스와 유사하나 인터페이스에서는 프로퍼티에 상태 정보를 저장할 수 없음

```kotlin
abstract class Vehicle // 추상클래스 정의 시 abstract 키워드 사용

abstract class Vehicle(val name: String, val color: String, val weight: Double) {
	abstract var maxSpeed: Double // 추상 프로퍼티 - 하위 클래스에서 재정의해 초기화해야함

	var year = "2021" // 일반 프로퍼티 - 초깃값 상태를 저장할 수 있음

	abstract fun start() // 추상 메소드이므로 하위 클래스에서 반드시 구현해야 함

	fun displaySpecs() { // 재정의 필요없음
		println("display spec")
	}
}
```

- 인터페이스 : abstract로 정의된 추상 메서드나 일반 메서드가 포함됨
    - 메서드에 구현 내용이 포함될 수 있으나 프로퍼티를 통해 상태를 저장할 수 없음

```kotlin
interface 인터페이스 이름 [: 인터페이스 이름 ...] {

	추상 프로퍼티 선언
	추상 메서드 선언	
	[일반 메서드 선언 { ... }]

}

interface Pet {
	var category: String // abstract 키워드가 없어도 기본은 추상 프로퍼티
	fun feeding() // 기본 추상 메소드
	fun patting() { // 일반 메서드 : 구현부를 포함하면 일반적인 메서드로 기본이 됨 
		println("Keep patting!") // 구현부
	}

	val msgTags: String // val 선언시 게터 구현 가능 
		get() = "I'm your lovely pet!"
}

class Cat(override var category: String) : Pet {
	override fun feeding() {
		println("Feed the cat a tuna can!")
	}
}

// 인퍼에스 다중 상속 가능

interface Bird {
	val wings: Int
	fun fly()
	fun jump() {
		println("bird jump!")
	}
}

interface Horse {
	val maxSpeed: Int
	fun run()
	fun jump() {
		println("horse jump!")
	}
}

class Pegasus: Bird, Horse { // 2개 인터페이스로부터 pegasus 클래스 정의
	override fun jump () {
		super<Horse>.jump()
		println("pegasus jump!")
	}
}
```

- 인터페이스 위임

```kotlin
// 메소드 접근

interface A {
	fun functionA() {}
}

interface B {
	fun functionB() {}
}

class Delegated(a: A, b: B) : A by a, B by b {
	fun functionC() {
		functionA() // . 점 없이 해당 메소드 접근 가능
		functionB()
	}
}

// 멤버 접근

interface Nameable {
	var name: String
}

class StaffName: Nameable {
	override var name: String = "Sean"
}

class Work: Runnable {
	override fun run() {
		println("work...")
	}
}

class Person(name: Nameable, work: Runnable): Nameable by name, Runnable by work

fun main() {
	val person = Person(StaffName(), Work()) // 생성자 사용해 객체 바로 전달
	println(person.name) // person -> staff -> name 접근
	person.run()
}
```

### 데이터 클래스와 기타 클래스

- Data class : 자바의 DTO 객체 기능 간략하게 표현 가능
    
    ```kotlin
    data class Customer(var name: String, var email: String)
    ```
    
    - equal(), hashCode(), toString(), copy(), component1(), component2() 등 자동 생성됨
    - 데이터 클래스 필수 조건
        - 주 생성자는 최소한 하나의 매개변수를 가져야 함
        - 주 생성자의 모든 매개변수는 val, var로 지정된 프로퍼티여야 함
        - 데이터 클래스는 abstract, open, sealed, inner 키워드 사용할 수 없음
        
        - 필요하다면 부 생성자나 init 블록을 넣어 데이터를 위한 간단한 로직 포함 가능
    - 데이터 클래스 자동 생성 메서드
        - equal() : 두 객체 내용이 같은지 비교하는 연산자 (고유 값은 다르지만 의미 값이 같을 때)
        - hashCode() : 객체를 구별하기 위한 고유한 정숫값 생성, 데이터 세트나 해시 테이블을 사용하기 위한 하나의 생성된 인덱스
        - copy() : 빌더 없이 특정 프로퍼티만 변경해서 객체 복사하기
        - toString() : 데이터 객체를 읽기 편한 문자열로 반환하기
        - componentN() : 객체의 선언부 구조를 분해하기 위해 프로퍼티에 상응하는 메서드
    - 객체 디스트럭처링 : 객체가 가지고 있는 프로퍼티를 개별 변수로 분해하여 할당하는 것
        - 변수를 선언할 때 소괄호를 사용해서 분해하고자 하는 객체를 지정한다
    
    ```kotlin
    val (name, email) = cus1
    // cus1 객체의 프로퍼티 값 2개를 각각 name과 email로 선언된 변수에 가져옴
    println("$name, $email") 
    
    println(_, email) = cus1 // 프로퍼티를 가져올 필요가 없는 경우 _ 사용해 제외 가능
    
    // 개별적으로 프로퍼티 가져오기 위해 componentN() 사용
    
    val name2 = cus1.component1()
    val email2 = cus1.component2()
    
    println("name = $name2, email = $email2") 
    
    // for 문 활용하여 모든 객체의 프로퍼티 분해 가능
    for((name, email) in customers) {
    	println("name = $name, email = $email")
    }
    
    // 람다식 활용하여 화면에 출력 가능
    val myLambda = {
    	(nameLa, emailLa) : Customer ->
    	println(nameLa)
    	println(emailLa)
    }
    
    myLambda(cus1)
    ```
    

### 내부 클래스

- 내부 클래스로 두는 이유
    - 독립적인 클래스로 정의하기 모호한 경우나
    - 다른 클래스에서는 사용하지 않는 내부에서만 사용하고 외부에서는 접근할 필요가 없는 경우
- 중첩 클래스 (nested) : 클래스 안에 다른 클래스가 정의되어 있는 것
    - static 처럼 다루어짐 : 객체 생성 없이 접근 가능

```kotlin
class Outer {
	val ov = 5
	class Nested {
		val nv = 10
		fun greeting() = "[Nested] Hello ! $nv" // 외부 ov에는 접근 불가
	}

	fun outside() {
		val msg = Nested().greeting() // 객체 생성 없이 중첩 클래스의 메서드 접근
		println("[Outer]: $msg, ${Nested().nv}") // 중첩 클래스의 프로퍼티 접근
	}
}

fun main() {
	// static 처럼 객체 생성 없이 사용 
	val output = Outer.Nested().greeting()
	println(output)

	val outer = Outer()
	outer.outside()
}
```

- 이너 클래스 : 이너 클래스는 바깥 클래스의 멤버들에 접근 가능

```kotlin
class Smartphone(val mode: String) {
	private val cpu = "Exynos"
	inner class ExternalStorage(val size: Int) {
		fun getInfo() = "${model}: Installed on $cpu with ${size}Gb"
	}
}

fun main() {
	val mySdCard = Smartphone("S7").ExternalStorage(32)
	println(mySdCard.getInfo())
}

// S7: Installed on Exynos with 32Gb
```

- 지역 클래스 : 특정 메서드의 블록이나 init 블록과 같이 블록 범위에서만 유효한 클래스

```kotlin
class Smartphone(val model: String) {
	fun powerOn(): String {
		class Led(val color: String) {
			fun blink(): String = "Blinking $color on $model"
		}
		val powerStatus = Led("Red")
		return powerStatus.blink()
	}
}

fun main() {
	val myphone = Smartphone("note9")
	myphone.ExternalStorage(128)
	println(myphone.powerOn())
}

// Blinking Red on Note9
```

- 실드 클래스 (sealed) : 미리 만들어둔 자료형 묶어서 제공 → 열거형의 확장

```kotlin
sealed class Result { // 실드 클래스 선언
	open class Success(val message: String): Result()
	class Error(val code: Int, val message: String): Result()
}

class Status: Result() // 상속은 같은 파일에서만 가능
class Inside: Result.Success("Status") // 내부 클래스 상속

//case 2

sealed class Result
open class Success(val message: String): Result()
class Error(val code: Int, val message: String): Result()

class Status: Result()
class Inside: Success("Status") // . 표기 없이 그대로 접근

// 사용 예

fun eval(result: Result): String = when(result) {
	is Status -> "in progress"
	is Result.Success -> result.message
	is Result.Error -> result.message
}

```

- 어노테이션 클래스
    - 어노테이션 - 코드에 부가 정보를 추가하는 역할 @ 기호와 함께 사용
    - 어노테이션 속성
        - @Target : 어노테이션이 지정되어 사용할 종류 (클래스, 함수, 프로퍼티 등) 정의
        - @Retention : 어노테이션을 컴파일된 클래스 파일에 저장할 것인지 실행 시간에 반영할 것인지 결정
        - @Repeatable : 어노테이션을 같은 요소에 여러 번 사용 가능하게 할지를 결정
        - @MustBeDocumented : 어노테이션이 API의 일부분으로 문서화하기 위해 사용

```kotlin
annotation class 어노테이션명 // 선언

annotation class Fancy // 선언
@Fancy class MyClass { ... } // 사용

// 속성과 함께 정의된 어노테이션 클래스 예 
@Target(AnnotationTarget.CLASS, AnnotationTarget.FUNCTION, 
	AnnotationTarget.VALUE_PARAMETER, AnnotationTarget.EXPRESSION)
@Retention(AnnotationRetention.SOURCE)
@MustBeDocumented
annotation class Fancy

// 클래스 레퍼런스 
클래스명::class 
User::class

//어노테이션 매개변수 생성자
annotation class Special(val why: String) // 어노테이션 클래스 매개변수 추가해서 정의
@Special("example") class Foo {} // 어노테이션에 매개변수 지정

```

### 연산자 오버로딩

- 연산자를 함수를 호출하는 식으로 사용 가능함 → 오버로딩도 가능

```kotlin
class Pointer(var x: Int = 0, var y: Int = 10) {
	operator fun plus(p: Point) : Point {
		return Point(x + x.p, y + p.y)
	}
}

fun main() {
	val p1 = Point(3, -8)
	val p2 = Point(2, 9)

	var point = Point()
	point = p1 + p2 // 연산자 오버로딩함 
	println("point = (${point.x}, ${point.y})")
```

- 산술 연산자
    - a * b : a.times(b)
    - a / b : a.div(b)
    - a % b : a.rem(b)
    - a..b : a.rangeTo(b)
- 호출 연산자 : 함수의 호출을 돕는데 사용

```kotlin
class Manager {
	operator fun invoke(value: String) {
		println(value)
	}
}

fun main() {
	val manager = Manager()
	manager("Do something for me!") // manager.invoke() 형태로 호출되며 invoke가 생략됨
}
```

- 인덱스 접근 연산자
    - a[i] = a.get(i)
    - a[i, j] = a.get(i, j)
    - a[i_ 1, ..., i_n] = a.get(i_1. ... i_n)
    - a[i] = b   =   a.set(i, b)
    - a[i, j] = b    = a.set(i, j, b)
    - a[i_1, ..., i_n] = b    =  a.set(i_1, ..., i_n, b)
- 단일 연산자 : a의 자료형을 결정하고 매개변수없이 각 연산자에 대한 함수를 호출한 다음 연산된 결과를 반환
    - +a = a.unaryPlus()
    - -a = a.unaryMinus()
    - !a = a.not()
- 범위 연산자 : in
    - a in b = b.contains(a)
    - a !in b = !b.contains(a)
- 대입 연산자 : 연산의 결과 할당
    - a += b    = [a.plus](http://a.plus)Assign(b)
    - a -= b     = a.minusAssign(b)
    - a *= b     = a.timesAssign(b)
    - a /= b     = a.divAssign(b)
    - a %= b    = a.remAssign(b)
- 동등성 연산자 : 일치와 불일치 판별
    - a == b    = a?equals(b)?: (b === null)
    - a ! = b    = !(a?equals(b)?: (b === null))
- 비교 연산자
    - a > b  = a.compareTo(b) > 0
    - a < b = a.compareTo(b) < 0
    - a >= b   = a.compareTo(b) >= 0
    - a <= b  = a.compareTo(b) <= 0
