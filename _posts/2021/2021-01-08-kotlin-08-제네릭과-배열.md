### 필기

### 제네릭

- 클래스 내부에서 사용할 자료형을 나중에 인스턴스를 생성할 때 확정함
    - 자료형의 객체들을 다루는 메서드나 클래스에서 컴파일 시간에 자료형을 검사해 적당한 자료형을 선택할수 있게 함
    - 객체의 자료형을 컴파일할때 체크하기 때문에 객체 자료형의 안정성을 높이고 형 변환의 번거로움이 줄어듬
- 앵글 브래킷 (<>) 안에 형식 매개변수를 넣어 선언하며 이 때 형식 매개변수는 하나 이상 지정할 수 있음
    - 형식 매개변수는 자료형을 대표하는 용어로 T 등 영문 대문자로 사용하며 나중에 필요한 자료형으로 대체됨
    - 다양한 자료형을 다루는 컬렉션에서 주로 사용됨

```kotlin
class Box<T>(t: T) { // T = 형식 매개변수명
	var name = t
}

fun main() {
	val box1: Box<Int> = Box<Int>(1) // 컴파일 시 지정
	val box2: Box<String> = Box<String>("Hello")
}
```

- 제네릭에서 사용하는 형식 매개변수명
    - E : element (요소)
    - K : key
    - N : number
    - T : type
    - V : value
    - S, U, V etc : 두 번째, 세 번째, 네 번째 형식
- 제네릭 클래스 : 형색 매개변수를 1개 이상 받는 클래스

```kotlin
class MyClass<T> {
	fun myMethod(a: T) { // 메서드의 매개변수 자료형에 사용됨
	}

	var myProp: T // 프로퍼티는 사용할 수 없음 - abstract로 선언해서 추상화해야함
}

// 주 생성자나 부 생성자에 형식 매개변수 지정해서 사용 가능

class MyClass<T> (val myProp: T) {}
class MyClass<T> {
	val myProp: T // 프로퍼티에서 사용
	constructor(myProp: T) { // 부 생성자 이용
		this.myProp = myProp
	}
}

// 객체 인스턴스 생성 시 명시적으로 자료형 이용 가능

var a = MyClass<Int>(12) // 주 생성자 myProp에 12할당되며 int 형으로 결정됨
println(a.myProp) // 12
```

- 자료형 변환 : 제네릭 클래스는 가변성을 지정하지 않으면 형식 매개변수에 상, 하위 클래스가 지정되어도 자료형이 변환되지 않음

```kotlin
open class Parent
class Child: Parent()

class Cup<T>

fun main() {
	val obj1: Parent = Child() // child는 parent 상속 받았기때문에 변환 가능
	val obj2: Child = Parent() // 오류! 

	val obj3: Cup<Parent> = Cup<Child>() // 오류! 자료형 불일치
	val obj4: Cup<CHild> = Cup<Parent>() // 오류! 자료형 불일치
	
	val obj5 = Cup<Child>() // 
	val obj6 = Cup<Child>() = obj5 // 자료형 일치하므로 가능
}
```

- 형식 매개변수의 null 제어 : 기본적으로 null이 가능

```kotlin
class GenericNull<T> {
	fun equalityFunc(arg1: T, arg2: T) {
		println(arg1?.equals(arg2))
	}
}

fun main() {
	val obj = GenericNull<Int?>() // null 가능한 형식 선언
	obj.equalityFunc(null, 10)
}
```

- null을 허용하지 않으려면 자료형 Any로 지정

```kotlin
class GenericNull<T:Any> {
}

fun main() {
	val obj2 = GenericNull<Int?>() // 오류! null 허용 안함
}
```

- 제네릭 함수 혹은 메서드 : 형식 매개변수를 받는 함수나 메서드

fun <형식 매개변수[,...]> 함수 이름(매개변수: <매개변수 자료형>[,...]): <반환 자료형>

```kotlin
fun <T> genericFunc(arg: T): T? { ..} // 매개변수와 반환 자료형에 형식 매개변수 T 사용
fun <K, V> put(key: K, value: V): Unit { ..} // 형식 매개변수가 2개인 경우

// 매개변수로 Array 사용하는 예

fun<T> find(a: Array<T>, Target: T): Int {
	for (i in a.indices) {
		if(a[i] == Target) return 1
	}
return -1
}
```

- 제네릭과 람다식 : 형식 매개변수로 선언된 매개변수 연산할 경우 자료형 불가 → 오류
    - 람다식을 매개변수로 받아 해결

```kotlin
fun <T> add(a: T, b: T): T { // 오류! 자료형 결정 불가
	return a + b
}

fun <T> add(a: T, b: T, op: (T, T) -> T): T {
	reutrn op(a, b)
}

fun main() {
	val result = add(2, 3, {a, b -> a + b})
	println(result)
}
```

- 다수 조건 형식 매개변수 제한 : where 사용

```kotlin
interface InterfaceA
interface InterfaceB

class HandlerA: InterfaceA, InterfaceB
class HandlerB: InterfaceA

// 2개의 인터페이스를 구현하는 클래스로 제한
class ClassA<T> where T: InterfaceA, T: InterfaceB

fun main() {
	val obj1 = ClassA<HandlerA>()
	val obj2 = ClassA<HandlerB>() // 범위에 안 맞음 -> 오류!
} 

// 함수에서 조건 제한
fun <T> myMax(a: T, b: T): T where T: Number, T: Comparable<T> {
}
```

- 상, 하위 형식 가변성 : 형식 매개변수가 클래스 계층에 영향을 주는 것
    - 형식 A의 값이 필요한 모든 클래스에 형식 B의 값을 넣어도 아무 문제가 없다면 B는 A의 하위 형식
    - 하위 클래스는 상위 클래스가 수용할 수 있음
- 무변성 : C<T> 와 C<T'>는 아무 관계가 없음
    - 형식 매개변수에 in이나 out 등으로 공변성이나 반공변성을 따로 지정하지 않으면 무변성으로 제네릭 클래스가 선언됨

```kotlin
class Box<T> (val size: Int) // 무변성 선언

fun main() {
	val anys: Box<Any> = Box<Int>(10) // 오류! 자료형 불일치
	val nothings: Box<Nothing> = Box<Int>(20)
}
```

- 공변성 : 형식 매개변수의 상하 자료형 관계가 성립하고, 그 관계가 그대로 인스턴스 자료형 관계로 이어지는 경우

```kotlin
class Box<out T>(val size: Int) // 공변성 선언

fun main() {
	val anys: Box<Any> = Box<Int>(10) // 관계 성립으로 객체 생성 가능
	val nothings: Box<Nothing> = Box<Int>(20) // 오류! nothing은 int의 하위 자료형이 아님
}

//out을 사용하는 경우 프로퍼티는 var로 지정될 수 없음 - var을 사용하려면 private만 가능
class Box<out T: Animal>(private var elem: T)
```

- 반공변성 : 자료형의 상하관계가 반대되어 인스턴스의 자료형이 상위 자료형이 됨

```kotlin
class Box<in T> (val size: Int) // 반공변성 선언
fun main() {
	val anys: Box<Any> = Box<Int>(10) // 자료형 불일치
	// 관계 성립 nothing 자료형의 상위 자료형이 int라 성립됨

	val nothings: Box<Nothing> = Box<Int>(20) 
} 
```

- 자료형 프로젝션 : 사용 지점 변성
    - out : 형식 매개변수는 세터를 통해 값을 설정하는 것이 제한됨
    - in : 형식 매개변수는 게터를 통해 값을 읽거나 반환할 수 있음

```kotlin
class Box<T>(var item: T) // 무변성

fun <T> printObj(box: Box<out Animal>) {
	val obj: Animal = box.item // item의 값을 얻음
	println(obj)
}
```

- 스타 프로젝션 : * 어떤 자료형이라도 들어올 수 있으나 구체적으로 자료형이 결정되고 난 후에는 그 자료형과 하위 자료형의 요소만 담을 수 있도록 제한할 수 있음
    - in, out은 사용 방법에 따라 결정됨

```kotlin
class InOutTest<in T, out U>(t: T, u: U) {
	val propT: T = t // 오류 out위치에 사용 불가
	val propU: U = u 

	fun func1(u: U) // 오류 in 위치에 사용 불가
	fun func2(t: T) {
		print(t)
	}
}

fun starTestFunc(v: InOutTest<*, *>) {
	v.func2(1) // 오류! nothng으로 인자 처리
	print(v.propU)
}
```

- reified : 매개변수로 지정하지 않고 형식 매개변수를 실행시간에 접근할 수 있음

```kotlin
inline fun <reified T> myGenericFun() // T를 일반 클래스처럼 사용할 수 있음
// inline 자료형은 인라인 함수에서만 사용됨 
```

### 배열 다루기

- 배열 : 동일한 자료형의 데이터를 연속적으로 나열한 형태
- arrayOf() 나 Array() 생성자를 통해 배열 생성 가능
- arrayOfNulls() 사용하여 빈 상태의 배열 생성 가능

```kotlin
val numbers = arrayOf(4, 5, 7, 3) // 정수형으로 초기화된 배열
val animals = arrayOf("Cat", "Dog", "Lion")

for (element in numbers) {
}
```

- 다차원 배열 : 기본 배열을 묶어 1차원 배열로 나타낸것

```kotlin
val array1 = arrayOf(1, 2, 3)
val array2 = arrayOf(4, 5, 6)
val array3 = arrayOf(7, 8, 9)

val arr2d = arrayOf(array1, array2, array3)
```

- 여러가지 자료형 혼합하여 사용 가능

```kotlin
val intOnlyArr1 = arrayOf<Int>(4, 5, 6, 7) // arrayOf<자료형> 형태로 자료형 제한 가능
```

- get, set 사용하여 배열의 값 접근 가능

```kotlin
arr.get(index) -> value = arr[index]
arr.set(index) -> arr[index] = value
```

- 배열 함수
    - arr.size : 배열의 크기
    - arr.sum() : 배열의 합
    - Arrays.toString(arr) : 배열의 내용을 문자열로 변환
    - Arrays.deepToString(arr) : 2차원 배열 문자열로 변환
    - arr.first() : 배열의 첫번째 요소 확인
    - arr.last() : 배열의 마지막 요소 확인
    - arr.indexOf(n) : 요소 n의 인덱스 출력
    - arr.average() : 배열의 평균 값 출력
    - arr.count() : 요소 개수 세기
    - arr.contains(n) : 배열에 요소 n이 포함되었는지 확인
    - arr.forEach() : 요소 개수만큼 지정한 구문 반복 실행
    
    ```kotlin
    arr.forEach {element -> print("$element")}
    // 순환하면서 인덱스도 출력
    arr.forEachIndexed({i, e -> println("arr[$i] = $e")}) 
    ```
    
    - arr.iterator() : 반복 요소 처리
    
    ```kotlin
    val iter: Iterator<Int> = arr.iterator()
    while(iter.hasNext()) {
    	val e = iter.next()
    }
    ```
    
    - arr.sortedArray(), arr.sortedArrayDescending() : 오름차순, 내림차순으로 정렬된 일반 배열로 반환
    - arr.sort(n, nn) : 인덱스 n부터 nn까지 정렬
    - arr.sortBy { it.price} : 값에 따라 정렬
    - sortWith() : 비교자로 정렬
    
    ```kotlin
    products.sortWith ( // comparaotr로 두 객체 비교
    	Comparator<Product> {p1, p2 ->
    		when {
    			}
    	}
    }
    ```
    
    - compareBy({조건}, {조건}) : 1차 조건으로 정렬 후 2차 조건으로 동일 내용 정렬
    - filter : 원하는 데이터 골라내기
    
    ```kotlin
    arr.filter { e -> e > 0}.forEach{ e-> print("$e")}
    ```

### 문자열 다루기

- 문자열 = 문자의 배열 : str[index] 형식으로 해당 문자 요소 접근 가능
- 문자열 추출 : str.subString(인덱스 범위 지정), charSequence.subSequence(인덱스 범위 지정)
- 문자열 비교 : str.compareTo(str2) → str과 str2 가 같으면 0반환
- StringBuilder : 문자열이 사용할 공간을 더 크게 잡을 수 있기 때문에 요소를 변경할 때 특정 단어를 변경할 수 있게 함

```kotlin
var s = StringBuilder("Hello")
s[2] = 'x' // 변경 가능

s.append("world") // 문자열 추가
s.insert(10, "added") // 인덱스 10번부터 추가됨
s.delete(5, 10) // 인덱스 5번부터 10번까지 삭제됨 
```
