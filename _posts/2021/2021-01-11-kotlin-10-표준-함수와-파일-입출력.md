### 코틀린 표준 함수

- 코드 단순화, 읽기 쉽게 함
- 람다식

```kotlin
val 변수 이름: 자료형 선언 = { 매개변수 [, ...] -> 람다식 본문}

val sum: (Int, Int) -> Int = { x, y -> x + y}
val mul = {x: Int, y: Int -> x * y}

// 매개변수 하나일 때 생략 후 it으로 표기 가능
val add: (Int) -> Int = {it + 1}

// 추론된 반환 자료명이 Unit이 아닐 경우 본문의 마지막 표현식이 반환값으로 처리됨
val isPositive: (Int) -> Boolean = {
	val isPositive = it > 0
	isPositive
}
```

- 고차 함수 : 함수의 매개변수로 함수를 받거나 함수 자체를 반환할 수 있는 함수

```kotlin
fun inc(x: Int): Int {
	return x + 1
}

fun high(name: String, body: (Int) -> Int): Int {
	println("name: $name")
	val x = 0
	return body(x)
}

// 함수 이용 람다식
val result = high("Sean", {x -> inc(x + 3)})
// 소괄호 바깥으로 빼내고 생략
val result2 = high("Sean") {inc (it + 3)}
// 매개변수 없이 함수의 이름만 사용할 떄
val result3 = high("Kim", ::inc)
// 람다식 자체를 넘겨준 형태
val result4 = high("Sean") { x -> x + 3}
// 매개변수가 1개인 경우 생략 
val result5 = high("Sean" { it + 3}
```

- 클로저 : 람다식으로 표현된 내부 함수에서 외부 범위에 선언된 변수에 접근할 수 있는 개념
    - 람다식 안에 있는 외부변수는 값을 유지하기 위해 람다식이 포획한 변수라고 통칭 : 참조가 유지되어 함수가 종료되어도 사라지지 않고 함수의 변수에 접근하거나 수정할 수 있게 함
    - final 변수를 포획한 경우 변수 값을 람다식과 함께 저장함
    - final이 아닌 변수를 포획한 경우 변수를 특정 래퍼로 감싸서 나중에 변경하거나 읽을 수 있게 함 → 래퍼에 대한 참조를 람다식과 함께 저장함

```kotlin
fun main() {
	val calc = Calc()
	var result = 0
	calc.addNum(2, 3) { x, y -> result = x + y}
	println(result)
}

class Calc {
	fun addNum(a: Int, b: Int, add: (Int, Int) -> Unit) {
		add(a, b)
	}
}
```

- let() : 함수를 호출하는 객체 T를 이어지는 block의 인자로 넘기고 block의 결괏값 R 반환
    - 메서드 체이닝 사용 가능

```kotlin
public inline fun <T, R> T.let(block: (T) -> R): R { ... return block(this)}

val score: Int? = 32

fun checkScoreLet() {
	// score의 멤버 메서드 호출하듯 let 호출 - 람다식 하나일때는 () 소괄호 생략
	score?.let {println("Score: $it")} // null 이 아닐 경우 {} 작업 수행
	// toString() 사용해 it을 문자열로 변환한 뒤 반환한 값 str에 할당함 
	val str = score.let{it.toString()}
	println(str)
}

// 메서드 체이닝 사용 예시

var a = 1
var b = 2 
a = a.let { it + 2}.let {
	val i = it + b
	i
}
```

- also() : 함수를 호출하는 객체 T를 이어지는 block에 전달하고 객체 T 자체를 반환
    - 블록 안의 코드 수행 결과와 상관 없이 T인 객체 this 반환

```kotlin
public inline fun <T> T.also(block: (T) -> Unit): T { block(this); return this}

var m = 1
m = m.also {it + 3}
println(m) // 1 : m의 기존 값이 1임 - 블록 내부 코드는 관련 없음 객체만 반환 
```

- apply() : 호출하는 객체 T를 이어지는 block안으로 전달하고 객체 자체인 this 반환

```kotlin
public inline fun <T> T.apply(block: T.() -> Unit): T { block(); return this}

data class Person(var name: String, var skills: String)
var person = Person("kildong", "kotlin")
	// 람다식이 확장 함수로 처리됨 
	person.apply {this.skills = "Swift"} // this 생략 가능
println(person) //skils : Swift
```

- run() : 인자가 없는 익명 함수처럼 동작하는 형태, 객체에서 호출하는 형태 2가지 사용 가능

```kotlin
public inline fun <R> run(block: () -> R): R = return block()
public inline fun <T, R> T.run(block: T.() -> R): R = return block()

val a = 10
var skills = "kotlin"

skils = run {
	val revel = "kotlin level:" + a
	level // 마지막 표현식이 반환됨
```

- with() : 인자로 받는 객체를 이어지는 block의 receiver로 전달하며 결괏값 반환

```kotlin
public inline fun <T, R> with(receiver: T, block: T.() -> R): R = receiver.block()

data class User(val name: String, var skills: String, var email: String ?= null)
val user = User("kildong", "default")

val result = with(user) { // 멤버 이름으로 접근 가능 
	skills = "Kotlin"
	email = "kildong@example.com"
	// "success" 마지막 표현식 반환 가능 -> 이 경우 result는 String 변수 됨
}
```

- use() : 객체를 사용한 후 close() 함수를 자동적으로 호출해 닫음 오류 여부 무관 - 항상 close() 호출 보장

```kotlin
public inline fun <T : Closable?, R> T.use(block: (T) -> R): R

// 파일 출력 후 hello 출력 
PrintWriter(FileOutputStream("d:\\test\\output.txt")).use {
	it.println("hello")
}
```

- takeIf() : 람다식이 true면 결과 반환
- takeUnless() : 람다식이 false면 결과 반환

```kotlin
public inline fun <T> T.takeIf(predicate: (T) -> Boolean): T?
	= if (predicate(this)) this else null

someObject?.takeIf {if.status}?.apply {doThis()} 

input.indexOf(keyword).takeUnless { it < 0} ?: error("keyword not found")
```

- Kotlin 입출력 API : [java.io](http://java.io) 라이브러리 → 파일이나 콘솔과 같은 스트림에서 읽거나 쓸 수 있는 API 제공
    - 간단한 데이터 : readBytes, readLInes, readText
    - 대량의 데이터 : copyTo, forEachBlock, forEachLine
    - 호출 후 종료 필요 : InputStream, Reader, Writer
- [java.io](http://java.io) vs java.nio
    - io → 스트림 방식 / nio → 채널 방식
    - io → Non-buffer / nio → buffer
    - io → 비동기 지원 X / 비동기 지원 (넌 블로킹 지원)
- 스트림 VS 채널
    - 스트림 : 데이터가 흘러가는 방향성에 따라 입력 스트림과 출력 스트림 구분 (데이터를 읽고 저장하는 양방향성 작업 시 input, output 두 작업 별도 지정 필요
    - 채널 : 양방향으로 입력, 출력 모두 가능하기 때문에 입출력 별도 지정 필요 없음
- 넌버퍼 vs 버퍼
    - 스트림 : 1바이트 쓰면 1바이트 읽음 → 버퍼로 읽는것보다 느림
- 블로킹 vs 넌블로킹
    - 블로킹 : write 작업 하려는데 공간이 없으면 대기 or read 작업 하려는데 읽을 내용이 없으면 대기
    - 넌블로킹 : 입출력 작업시 스레드나 비동기 루틴에 맡겨 별개의 흐름으로 작업하게 되는것
- 파일에 쓰기
    - Standard open option
        - READ : 파일을 읽기용으로 오픈
        - WRITE : 파일을 쓰기용으로 오픈
        - APPEND : 파일이 존재하면 마지막에 추가함
        - CREATE : 파일이 없으면 새 파일을 생성

```kotlin
// Files.write 사용
val path = "D:\\" // 파일 생성 경로
val text = "안녕하세요!" // 파일에 쓸 내용

try {
	Files.write(Paths.get(path), text.toByteArray(), StandardOpenOption.CREATE)
} catch (e: IOException) {
}

// PrintWriter 사용
val outString: String = "안녕하세요" // 입력할 내용
val path = "D:\\" // 저장 경로

val file = File(path)
val printWriter = PrintWriter(file)

printWriter.println(outString) // 파일에 출력
printWriter.close() 
// 자동으로 close 하도록 use 사용시
File(path).printWriter().use { it.println(outString)} 

// BufferedWriter 사용
File(path).bufferedWriter().use {it.write(outString)}

// File writeText() 사용
val file = File(path)
file.writeText(outString)
file.appendText("do great work!")

// FileWriter 사용
val writer = FileWriter(path, true) // 경로, append 여부
try { 
	writer.write(outString)
} catch (e: Exception) {
} finally {
	writer.close()
}

FileWriter(path, true).use { it.write(outString)} 
```

