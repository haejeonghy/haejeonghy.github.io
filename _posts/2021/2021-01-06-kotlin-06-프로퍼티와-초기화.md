### 프로퍼티의 접근

- 코틀린에서는 변수 선언 부분과 기본적인 접근 메서드를 모두 가지고 있음
- 직접 선언할 수 있음

```kotlin
class User(_id: Int, _name: String, _sage: Int) {
	val id: Int = _id
			get() = field
	var name: String = _name
			get() = field
			// private set 등으로 가시성 지시자 추가하면 외부에서 프로퍼티에 값을 할당할 수 없음
			set(value) { //value : 세터의 매개변수로 외부로부터 값을 가져옴
				field = value  // field : 프로퍼티를 참조하는 변수
				// value로 받은 변수를 변환 (예 - 대소문자 변경 등) 할 수 있어 편리함
			}
	var age: Int = _age
			get() = field
			set(value) {
				field = value
			}

	// 임시적인 보조 프로퍼티 
	private var tempName: String? = null

	// open 사용 시 오버라이딩 가능함
	open val x: Int = 0

}
```

### 지연 초기화와 위임

- 지연 초기화 : 객체의 정보가 나중에 나타나는 경우 객체 생성과 동시에 초기화하기 힘듬 → 이때 지연 초기화 사용
- lateinit : 프로퍼티에 값이 바로 할당되지 않아도 컴파일러에서 허용
    - var로 선언된 프로퍼티만 가능함
    - 프로퍼티에 대한 게터와 세터를 사용할 수 없음

```kotlin
class Person {
	lateinit var name: String // 초기화 하지 않은 채 선언 가능
}

// 객체 지연 초기화 

data class Person(var name: String, var age: Int)
late init var person1: Person

fun main() {
	person1 = Person("kildong", 30)
}
```

- lazy : 읽기 전용의 val 객체나 프로퍼티 나중에 초기화할 때 사용
    - 값을 다시 변경할 수 없음
    - 호출 시점에 by lazy {...} 정의에 의해 블록 부분의 초기화 진행

```kotlin
class LazyTest {
	init {
	}

	val subject by lazy {
		"kotlin programming" // 반환값
	}

	fun flow() {
		println("$subject") // 최초 초기화
		println("$subject") // 이미 초기화된 값 사용
	}
}

fun main() {
	val test = LazyTest()
	test.flow()
}

```

- by : 하나의 클래스가 다른 클래스에 위임하도록 선언하여 위임된 클래스가 가지는 멤버를 참조 없이 호출할 수 있게 됨
    - 프로퍼티 위임 : 프로퍼티의 게터와 세터를 특정 객체에게 위임하고 그 객체가 값을 읽거나 쓸 때 수행하도록 만드는 것

```kotlin
< val | var | class > 프로퍼티 혹은 클래스 이름: 자료형 by 위임자

// 클래스의 위임
interface Animal {
	fun eat() { .. }
}

class Cat : Animal {}
val cat = Cat()
class Robot : Animal by cat // animal의 정의된 cat의 모든 멤버를 robot에 위임

// 예제

interface Car {
	fun go(): String
}

class VanImpl(val power: String): Car {
	override fun go() = "은 짐을 적재하며 $power 을 가집니다"
}

class CarModel(val mode: String, impl: Car): Car by impl {
	fun carInfo() {
		println("$model ${go()}") // 참조 없이 인터페이스 구현 클래스의 go()에 접근
	}
}

fun main() {
	val damas = CarModel("Damas 2010", VanImpl("100마력"))
	damas.carInfo()
}0
```

- observable() : 코틀린 Delegate 임포트 필요 - 프로퍼티를 감시하고 있다가 특정 코드의 로직에서 변경이 일어날 때 호출되어 처리됨 (콜백)

```kotlin
class User {
	var name: String by Delegates.observable("NONAME") {
		prop, old, new -> // 프로퍼티, 기존 값, 새로운 값
			println("$old -> $new") //변경 이벤트 발생할 때 실행됨
	}
}

fun main() {
	val user = User()
	user.name = "Kildong"
	user.name = "Dooly"
}
```

- vetoable() : 반환값에 따라 프로퍼티 변경을 허용하거나 취소할 수 있음

```jsx
fun main() {
	var max: Int by Delegates.vetoable(0) {
		prop, old, new -> 
		new > old
	}
	
	println(max)
	max = 10
	println(max)

	max = 5 // 기존의 값보다 작은 값이 할당되므로 거부됨
	println(max)
```

### 정적 변수와 컴패니언 객체

- 정적 변수, 컴패니언 객체 : 동적인 초기화 없이 사용가능한 변수 개념
- 컴패니언 객체 사용하기 : 정적 변수 사용시 static 키워드 사용하지 않고 컴패니언 객체 제공함

```kotlin
class Person {
	var id: Int = 0
	var name: String = "Youngdeok"
	companion object {
		var language: String = "Korean"
		fun work() {
			println("working...")
		}
	}
}

fun main() {
	println(Person.language) // 인스턴스 변경하지 않고 기본값 사용
	Person.language = "English" // 기본값 변경 가능
	Person.work() // 메서드 실행
	println(Person.name) // 컴패니언 객체 아니기 때문에 오류
}
```

- 최상위 함수 : 클래스 없이 만든 함수는 객체 생성 없이도 main() 함수 어디에서든 실행 가능함 → 역컴파일 해보면 static으로 선언되어있음
- object와 싱글톤

```kotlin
object OCustomer {
	var name = "kildong"
	fun greeting() = println("Hello World!")
	val HOBBY = hobby("Basketball")
	init {
		println("Init!")
	}
}

class CCustomer {
	companion object {
		const val HELLO = "hello"
		var name = "Joosol"
	}
}

fun main() {
	// . (점) 표기법으로 바로 사용가능 - 단일 인스턴스 생성하기 때문에 싱글톤 패턴으로 처리됨
	// object 사용 시 접근 시점에 객체가 생성되므로 생성자 호출을 하지 않고
	// , 주생성자 부생성자 사용 불가
	OCustomer.greeting()
	OCustomer.name = "Dooly"
}

// 표현식

open class Superman() {
	fun work() = println("work")
	fun talk() = println("talk")
	open fun fly() = println("fly")
}

fun main() {
	val pretendedMan = object: Superman() {
	override fun fly() = println("superman") // object 표현식으로 fly() 구현 재정의
	}

	pretendedMan.work()
}
```
